<#
.SYNOPSIS
    Minecraft Yasaklı Mod/Hile Kontrol Aracı
.DESCRIPTION
    Bu script, Minecraft oyun dizinini tarayarak yasaklı modları ve hileleri tespit eder.
.NOTES
    File Name      : Minecraft-Hile-Kontrol.ps1
    Author         : [İsminiz]
    Prerequisite   : PowerShell v5.1+
#>

# Yasaklı mod/hile listesi
$yasakliModlar = @(
    "Wurst", "Aristois", "Impact", "Sigma", "Huzuni",
    "Inertia", "Future", "Phobos", "KamiBlue", "WWE",
    "Bleach", "RusherHack", "Konas", "Pyro", "Abyss",
    "Lambda", "Mathax", "Seppuku", "Gamesense", "Salhack",
    "ZeroDay", "Vape", "Raven", "Exodus", "FDPClient",
    "Jigsaw", "W+", "LiquidBounce", "CheatBreaker", "Fenix"
)

# Minecraft mods klasörü yolu (varsayılan)
$minecraftModsPath = "$env:APPDATA\.minecraft\mods"

# Kontrol fonksiyonu
function KontrolEt {
    param (
        [string]$path
    )
    
    Write-Host "`nMinecraft Hile Kontrol Aracı" -ForegroundColor Cyan
    Write-Host "============================"
    Write-Host "`nTarama başlatılıyor...`n"
    
    if (-not (Test-Path $path)) {
        Write-Host "[HATA] Minecraft mods klasörü bulunamadı: $path" -ForegroundColor Red
        return
    }
    
    $modFiles = Get-ChildItem -Path $path -File -Recurse -ErrorAction SilentlyContinue
    
    if ($modFiles.Count -eq 0) {
        Write-Host "[BİLGİ] Mods klasörü boş veya mod bulunamadı." -ForegroundColor Green
        return
    }
    
    $yasakliBulunanlar = @()
    
    foreach ($file in $modFiles) {
        foreach ($yasakli in $yasakliModlar) {
            if ($file.Name -like "*$yasakli*") {
                $yasakliBulunanlar += $file.Name
                Write-Host "[YASAKLI MOD] $($file.Name)" -ForegroundColor Red
                break
            }
        }
    }
    
    if ($yasakliBulunanlar.Count -gt 0) {
        Write-Host "`n[UYARI] Toplam $($yasakliBulunanlar.Count) yasaklı mod/hile bulundu!" -ForegroundColor Yellow
        Write-Host "`nBulunan yasaklı modlar/hileler:`n"
        $yasakliBulunanlar | ForEach-Object { Write-Host "- $_" }
        
        # Ek işlemler (isteğe bağlı)
        Write-Host "`n[ÖNERİ] Bu modları hemen silmelisiniz!" -ForegroundColor Magenta
    }
    else {
        Write-Host "`n[TEBRİKLER] Hiçbir yasaklı mod/hile bulunamadı!" -ForegroundColor Green
    }
}

# Ana işlem
KontrolEt -path $minecraftModsPath

# Kullanıcı çıkış için beklesin
Write-Host "`nÇıkmak için bir tuşa basın..."
$null = $Host.UI.RawUI.ReadKey("NoEcho,IncludeKeyDown")
